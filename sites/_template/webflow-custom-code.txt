WEBFLOW CUSTOM CODE

Repo: MikeRudi/mharris-webflow


HEAD CODE

<script>
  (function () {
    window.mhLoadGithubFile = async function (path) {
      const url =
        "https://api.github.com/repos/MikeRudi/mharris-webflow/contents/" +
        path +
        "?ref=main&v=" +
        Date.now();

      const response = await fetch(url, { cache: "no-store" });
      if (!response.ok) throw new Error("GitHub load failed: " + path);

      const file = await response.json();
      return atob(file.content.replace(/\s/g, ""));
    };

    window.mhLoadGithubFile("sites/_template/src/styles.css")
      .then(function (css) {
        const style = document.createElement("style");
        style.setAttribute("data-mh-source", "styles.css");
        style.textContent = css;

        document.head.appendChild(style);
      })
      .catch(function (error) {
        console.error(error);
      });
  })();
</script>
