HEAD CODE

<script>
  (function () {
    window.mhWebflowVersion = Date.now();

    const mhStyles = document.createElement("link");
    mhStyles.rel = "stylesheet";
    mhStyles.href =
      "https://cdn.jsdelivr.net/gh/MikeRudi/mharris-webflow@main/sites/_template/src/styles.css?v=" +
      window.mhWebflowVersion;

    document.head.appendChild(mhStyles);
  })();
</script>
